143 {
144  switch(xed_state_get_machine_mode(p)) {
145  case XED_MACHINE_MODE_LONG_64:
146  return XED_ADDRESS_WIDTH_64b;
147 
148  case XED_MACHINE_MODE_REAL_16:
149  /* should be 20b... but if you are working w/real mode then you're
150  going to have to deal with somehow. Could easily make this be
151  20b if anyone cares. */
152  return XED_ADDRESS_WIDTH_32b;
153 
154  case XED_MACHINE_MODE_LEGACY_32:
155  case XED_MACHINE_MODE_LONG_COMPAT_32:
156  return XED_ADDRESS_WIDTH_32b;
157  case XED_MACHINE_MODE_LEGACY_16:
158  case XED_MACHINE_MODE_LONG_COMPAT_16:
159  return XED_ADDRESS_WIDTH_16b;
160  default:
161  return XED_ADDRESS_WIDTH_INVALID;
162  }
